    var int x;
    {
      x = 0;
      while (x < 11) do{
        Arreglo[x] = y * x;
        x = x + 1;
      }
    }

  var int i;
  {
    i=j+(p-j*2+j);

    if(j != 1) then{
      return (j);
    }
    else{
      return (j * 2);
    }
  }
  
main(){
